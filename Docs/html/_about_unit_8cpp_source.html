74 void __fastcall TAboutForm::SetAboutCaption()
75 {
76  DWORD VersionHandle;
77  DWORD VersionSize;
78  LPBYTE pBuffer;
79  UnicodeString strVersion = L"N/A";
80  UnicodeString NL = '\n';
81 
82  VersionSize = GetFileVersionInfoSizeW(Application->ExeName.c_str(), &VersionHandle);
83  if(VersionSize)
84  {
85  pBuffer = new BYTE[VersionSize];
86 
87  if(GetFileVersionInfoW(Application->ExeName.c_str(), VersionHandle, VersionSize, pBuffer))
88  {
89  VS_FIXEDFILEINFO *fi;
90  UINT buflen;
91 
92  //uncomment strVersion and HIWORD alternates below when future CI implemented: sas@2.1.0
93  if(VerQueryValueW(pBuffer, L"\\", (void** )&fi, &buflen))
94  {
95  //strVersion.sprintf(L"%d.%d.%d (Build %d)",
96  strVersion.sprintf(L"%d.%d.%d",
97  HIWORD(fi->dwFileVersionMS), LOWORD(fi->dwFileVersionMS),
98  HIWORD(fi->dwFileVersionLS)
99  //HIWORD(fi->dwFileVersionLS), LOWORD(fi->dwFileVersionLS)
100  );
101  }
102  }
103  delete[] pBuffer;
104  }
105 
106  AboutLabelCaption->Caption = L"All the tools to design, build and operate your own railway" + NL + NL +
107  L"Release: " + Interface->ProgramVersion + NL + NL +
108  L"Copyright (c) 2010 - 2023 Albert Ball\nwith contributions from Stephen A Smith and Kristian Zarębski";
