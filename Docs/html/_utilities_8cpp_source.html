131 void TUtilities::SaveFileString(std::ofstream &OutFile, AnsiString SaveString)
132  // If don't send a non- '\n' delimiter (using '\0' for convenience, tested in LoadFileString) then when string is null
133  // two '\n' characters are placed together. Then, during reloading, the second one is read because a readable character
134  // is expected, and on finishing the '\n' that would otherwise be the next character in the file isn't there. Using '\0'
135  // allows the '\n' to remain in place. Also see LoadFileString.
136 {
137  OutFile << SaveString.c_str() << '\0' << '\n';

356 bool TUtilities::CheckFileString(std::ifstream &InFile)
357  // Reads the next item and checks it as a string value up to either the '\0' delimiter
358  // if there is one, in which case the '\0' is extracted but nothing more, or up to the next '\n',
359  // in which case the '\n' is extracted. There may or may not be a '\n' at the start, and if there
360  // is it is ignored (only one is ignored, a second one is treated as a delimiter).
361 {
362  char TempChar;
363  InFile.get(TempChar); // may or may not be '\n'
364  if(InFile.fail())
365  return false;
366  if(TempChar == '\n')
367  {
368  InFile.get(TempChar); // get the next one if first was '\n'
369  if(InFile.fail())
370  return false;
371  }
372  while((TempChar != '\0') && (TempChar != '\n'))
373  {
374  if((TempChar < 32) && (TempChar >= 0))
375  return false;
376  InFile.get(TempChar);
377  if(InFile.fail())
378  return false;
379  }
380  return true;
381 }

413 bool TUtilities::CheckAndCompareFileString(std::ifstream &InFile, AnsiString InString)
414  // Reads the next item and checks it as a string value up to either the '\0' delimiter
415  // if there is one, in which case the '\0' is extracted but nothing more, or up to the next '\n',
416  // in which case the '\n' is extracted. There may or may not be a '\n' at the start, and if there
417  // is it is ignored (only one is ignored, a second one is treated as a delimiter).
418  // The item is then compared with InString and fails if different.
419 {
420  char TempChar;
421  char *Buffer = new char[10000];
422  int Count = 0;
423  InFile.get(TempChar); // may or may not be '\n'
424  if(InFile.fail())
425  {
426  delete Buffer;
427  return false;
428  }
429  if(TempChar == '\n')
430  {
431  InFile.get(TempChar); // get the next one if first was '\n'
432  if(InFile.fail())
433  {
434  delete Buffer;
435  return false;
436  }
437  }
438  while((TempChar != '\0') && (TempChar != '\n'))
439  {
440  if((TempChar < 32) && (TempChar >= 0))
441  {
442  delete Buffer;
443  return false;
444  }
445  Buffer[Count] = TempChar;
446  Count++;
447  InFile.get(TempChar);
448  if(InFile.fail())
449  {
450  delete Buffer;
451  return false;
452  }
453  }
454  Buffer[Count] = '\0';
455  Count++;
456  Buffer[Count] = '\n';
457  Count++;
458  if(AnsiString(Buffer) != InString)
459  {
460  delete Buffer;
461  return false;
462  }
463  delete Buffer;
464  return true;
465 }

520 AnsiString TUtilities::Format96HHMMSS(TDateTime DateTime)
521  // Formats a TDateTime into an AnsiString of the form hh:mm:ss where hh runs from 00 to 95 & resets when it reaches 96
522 {
523  AnsiString MinSecString = DateTime.FormatString(":nn:ss");
524  int Hours = (int)(((double)(DateTime + 0.000006)) * 24); // for v0.6 round up by ~0.5 sec to prevent undershooting
525  while(Hours >= 96)
526  Hours -= 96;
527  AnsiString HourString = AnsiString(Hours);
528  while(HourString.Length() < 2)
529  HourString = "0" + HourString;
530  return (HourString + MinSecString);
531 }
