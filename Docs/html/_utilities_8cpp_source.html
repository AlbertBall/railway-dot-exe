135 void TUtilities::SaveFileString(std::ofstream &OutFile, AnsiString SaveString)
136 // If don't send a non- '\n' delimiter (using '\0' for convenience, tested in LoadFileString) then when string is null

399 bool TUtilities::CheckFileString(std::ifstream &InFile)
400 // Reads the next item and checks it as a string value up to either the '\0' delimiter
401 // if there is one, in which case the '\0' is extracted but nothing more, or up to the next '\n',
402 // in which case the '\n' is extracted. There may or may not be a '\n' at the start, and if there
403 // is it is ignored (only one is ignored, a second one is treated as a delimiter).
404 {
405  char TempChar;
406  InFile.get(TempChar); // may or may not be '\n'
407  if(InFile.fail())
408  {
409  return(false);
410  }
411  if(TempChar == '\n')
412  {
413  InFile.get(TempChar); // get the next one if first was '\n'
414  if(InFile.fail())
415  {
416  return(false);
417  }
418  }
419  while((TempChar != '\0') && (TempChar != '\n'))
420  {
421  if((TempChar < 32) && (TempChar >= 0))
422  {
423  return(false);
424  }
425  InFile.get(TempChar);
426  if(InFile.fail())
427  {
428  return(false);
429  }
430  }
431  return(true);
432 }

474 bool TUtilities::CheckAndCompareFileString(std::ifstream &InFile, AnsiString InString)
475 // Reads the next item and checks it as a string value up to either the '\0' delimiter
476 // if there is one, in which case the '\0' is extracted but nothing more, or up to the next '\n',
477 // in which case the '\n' is extracted. There may or may not be a '\n' at the start, and if there
478 // is it is ignored (only one is ignored, a second one is treated as a delimiter).
479 // The item is then compared with InString and fails if different.
480 {
481  char TempChar;
482  char *Buffer = new char[10000];
483  int Count = 0;
484  InFile.get(TempChar); // may or may not be '\n'
485  if(InFile.fail())
486  {
487  delete[] Buffer;
488  return(false);
489  }
490  if(TempChar == '\n')
491  {
492  InFile.get(TempChar); // get the next one if first was '\n'
493  if(InFile.fail())
494  {
495  delete[] Buffer;
496  return(false);
497  }
498  }
499  while((TempChar != '\0') && (TempChar != '\n'))
500  {
501  if((TempChar < 32) && (TempChar >= 0))
502  {
503  delete[] Buffer;
504  return(false);
505  }
506  Buffer[Count] = TempChar;
507  Count++;
508  InFile.get(TempChar);
509  if(InFile.fail())
510  {
511  delete[] Buffer;
512  return(false);
513  }
514  }
515  Buffer[Count] = '\0';
516  Count++;
517  Buffer[Count] = '\n';
518  Count++;
519  if(AnsiString(Buffer) != InString)
520  {
521  delete[] Buffer;
522  return(false);
523  }
524  delete[] Buffer;
525  return(true);
526 }

618 bool TUtilities::ReadOneLineFromCouplingFile(std::ifstream &InFile, AnsiString &OutString)
619 {
620  OutString = "";
621  char TempChar;
622  char *Buffer = new char[1200];
623  int Count = 0;
624  InFile.get(TempChar); //should be a letter or eof
625  if(InFile.eof())
626  {
627  delete[] Buffer;
628  return(true);
629  }
630  if(InFile.fail())
631  {
632  delete[] Buffer;
633  return(false);
634  }
635  while(TempChar != ';') //reading first name
636  {
637  Buffer[Count] = TempChar;
638  Count++;
639  if(Count > 1000)
640  {
641  delete[] Buffer;
642  return(false);
643  }
644  InFile.get(TempChar);
645  if(InFile.fail())
646  {
647  delete[] Buffer;
648  return(false);
649  }
650  }
651  //here at semicolon after 1st name
652  Buffer[Count] = TempChar; //1st semicolon
653  Count++;
654  InFile.get(TempChar);
655  if(InFile.fail())
656  {
657  delete[] Buffer;
658  return(false);
659  }
660  while(TempChar != ';') //reading first exit ID
661  {
662  Buffer[Count] = TempChar;
663  Count++;
664  if(Count > 1000)
665  {
666  delete[] Buffer;
667  return(false);
668  }
669  InFile.get(TempChar);
670  if(InFile.fail())
671  {
672  delete[] Buffer;
673  return(false);
674  }
675  }
676  //here at semicolon after 1st exit ID
677  Buffer[Count] = TempChar; //add the semicolon
678  Count++;
679  InFile.get(TempChar);
680  if(InFile.fail())
681  {
682  delete[] Buffer;
683  return(false);
684  }
685  while(TempChar != ';') //reading 2nd name
686  {
687  Buffer[Count] = TempChar;
688  Count++;
689  if(Count > 1000)
690  {
691  delete[] Buffer;
692  return(false);
693  }
694  InFile.get(TempChar);
695  if(InFile.fail())
696  {
697  delete[] Buffer;
698  return(false);
699  }
700  }
701  //here at semicolon after 2nd name
702  Buffer[Count] = TempChar; //add the semicolon
703  Count++;
704  InFile.get(TempChar);
705  if(InFile.fail())
706  {
707  delete[] Buffer;
708  return(false);
709  }
710  while(TempChar != '-') //reading 2nd exit ID, look for the dash
711  {
712  Buffer[Count] = TempChar;
713  Count++;
714  if(Count > 1000)
715  {
716  delete[] Buffer;
717  return(false);
718  }
719  InFile.get(TempChar);
720  if(InFile.fail())
721  {
722  delete[] Buffer;
723  return(false);
724  }
725  }
726  //here at dash in 2nd exit ID
727  Buffer[Count] = TempChar; //add the dash
728  Count++;
729  InFile.get(TempChar); //1st digit after the dash
730  if(InFile.fail())
731  {
732  delete[] Buffer;
733  return(false);
734  }
735  while((TempChar >= '0') && (TempChar <= '9'))
736  {
737  Buffer[Count] = TempChar;
738  Count++;
739  if(Count > 1000)
740  {
741  delete[] Buffer;
742  return(false);
743  }
744  TempChar = InFile.peek(); //if a number, read it and continue, if a letter stop here (1st letter of 2nd line name), if a control character extract it and stop
745  if((TempChar >= '0') && (TempChar <= '9')) //number
746  {
747  InFile.get(TempChar);
748  if(InFile.fail())
749  {
750  delete[] Buffer;
751  return(false);
752  }
753  continue;
754  }
755  if(TempChar < ' ') //control character (less than space)
756  {
757  InFile.get(TempChar); //extract it, but may be last so check for eof
758  if(InFile.eof()) //if so then have a full buffer so need to define OutString
759  {
760  Buffer[Count] = '\0';
761  Count++;
762  Buffer[Count] = '\n';
763  Count++;
764  OutString = AnsiString(Buffer);
765  delete[] Buffer;
766  return(true);
767  }
768  if(InFile.fail())
769  {
770  delete[] Buffer;
771  return(false);
772  }
773  }
774  //anything other than a number or control character
775  }
776  //here when line fully defined in Buffer & file pointer at start of next line
777  Buffer[Count] = '\0';
778  Count++;
779  Buffer[Count] = '\n';
780  Count++;
781  OutString = AnsiString(Buffer);
782  delete[] Buffer;
783  return(true);
784 }
